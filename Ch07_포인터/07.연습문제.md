10. 다음 프로그램의 결과에 대해서 설명하라.
```C
1 #include <stdio.h> 
2 int main()
3 {
4 	int a[2][3]={ {-3, 14, 5}, {1, -10, 8} };
5 	int *b[]={a[0],a[1] };
6 	int *p=b[1];
7 	int ar[]={10, 15, 4, 25, 3, -4};
8 	int *q;
9 	q=&ar[2];
10 	printf("%d\n", *b[1]);
11 	printf("%d\n", *(++p));
12 	printf("%d\n", *(*(a+1)+1));
13 	printf("%d\n", *(--p-2));
14 	printf("%d\n", *(q+1));
15 	printf("%d\n", q[-1]);
16 	printf("%d\n", (ar-q));
17 	printf("%d\n", ar[*p++]);
18 	printf("%d\n", *(ar+ar[2]));
19 }
```
