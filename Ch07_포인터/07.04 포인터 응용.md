#### ● 예제 07-04-1 : 배열의 인덱스를 사용하는 방법과 포인터를 사용하는 방법의 비교
```C
1 #include <stdio.h>
2 #define ROWS 5
3 #define COLS 10
4 
5 int multi[ROWS][COLS];
6 
7 int main()
8 {
9 	int row, col;
10 	for (row = 0; row < ROWS; row++)
11 		for (col = 0; col < COLS; col++)
12 			multi[row][col] = row*col;
13 	for (row = 0; row < ROWS; row++) {
14 		for (col = 0; col < COLS; col++)
15 			printf("\n%d ", multi[row][col]);
16 		printf("\n");
17 	}
18 	printf("\n\n");
19 	for (row = 0; row < ROWS; row++) {
20 		for (col = 0; col < COLS; col++)
21 			printf("%3d ", *(*(multi + row) + col));
22 		printf("\n");
23 	}
24 }
```

#### ● 해설
* 2,3: 매크로를 이용하여 행과 열을 정의
* 5: 전역 변수로 배열 선언, 함수 밖에서 배열이나 변수를 선언 할 수 있다. 7장에서 자세히 볼 것이다.
* 10\~12: 배열의 행, 열에 데이터 저장
* 13\~17: 행렬에 저장된 값을 배열의 인덱스를 이용한 출력
* 18\~23: 행렬에 저장된 값을 배열의 포인터를 이용한 출력
> [!NOTE]
> 21행은 2차원 배열의 위치를 계산하는 방법을 보여준다. 다차원 배열일 때 그 위치를 정확하게 알려면 이와 같은 방법으로 찾을 수 있다.

#### ● 예제 07-04-2 : 2차원 정수 배열을 포인터의 포인터로 할당하는 방법 및 사용법

```C
13 	}
14 	printf("\n\n\nIndex Pointer(hex) Pointer(dec) Diff.(dec)");
```

* 16: 2차원 포인터 연결할 메모리 확보
* 17\~20; 포인터가 확보되지 않으면 에러메시지 출력하고 끝냄
* 21: 포인터에 대한 정보 출력, 행 번호 및 행의 주소를 16진수, 10진수로 출력

#### ● 예제 07-04-3 : 2차원 정수 배열을 포인터의 포인터로 할당하는 방법 및 사용법

```C
39 	puts("\n");
40 	/* 여기서는 2차원 배열이 메모리 블록에서 연속적으로 저장되어 있는지를

45 		for (col = 0; col < ncols; col++)

48 	}
49 	free(aptr);
50 	free(rptr);
```

* 15: 각 행의 포인터를 저장할 공간 확보

* 34: 각 행렬의 요소에 값을 저장한다.

#### ● 예제 07-04-4 : 3차원 배열 공간의 확보 방법
```C
1 #include <stdio.h>
2 #include <stdlib.h>
3 #include <stddef.h>
4 int X_DIM=16;
5 int Y_DIM=5;
6 int Z_DIM=3;
7 
8 int main( )
9 {
10 	char *space, ***Arr3D;
11 	int y, z;
12 	ptrdiff_t diff;
13 	/* 3차원 배열 전체에 대한 공간 확보 */
14 	space = malloc(X_DIM * Y_DIM * Z_DIM * sizeof(char));
15 	/* 포인터의 배열에 대한 공간 할당, 이들 각각은 결과적으로 2차원 배열의
16 	첫 번째 요소를 포인트한다. 즉, 포인터의 포인터 2차원 배열이다. */
17 	Arr3D = malloc(Z_DIM * sizeof(char **));
18 	/* 행에 대한 포인터 배열을 새로 할당된 공간에 대입한다. */
19 	for (z = 0; z < Z_DIM; z++) {
20 		Arr3D[z] = malloc(Y_DIM * sizeof(char *));
21 		/* 이들 각각의 공간에는 각 행의 첫 번째 요소를 포인트한다. 
22 		즉, 원래 할당된 각 행의 배열 공간에 있는 첫 번째 요소이다. */
23 		for (y = 0; y < Y_DIM; y++) {	// 2차원 배열의 시작주소 계산
24 			Arr3D[z][y] = space + (z*(X_DIM * Y_DIM) + y*X_DIM);
25 		}
26 	}
27 	/* 이제, 3차원 공간이 연속적으로 어떻게 저장되는지 배열의 인덱스를
28 	통해서 3차원 배열공간의 각 주소를 점검해보자. */
29 	for (z = 0; z < Z_DIM; z++) {
30 		printf("배열 %d의 위치는 %p\n", z, *Arr3D[z]);
31 		for ( y = 0; y < Y_DIM; y++) {
32 			printf(" 배열 %d, 행 %d의 시작주소: %p", z, y, 
33 				Arr3D[z][y]);
34 			diff = Arr3D[z][y] - space;
35 			printf(" diff = %4d ", diff);
36 			printf(" z = %d y = %d\n", z, y);
37 		}
38 	}
39 	free(space);
40 	free(Arr3D);
41 }
```

#### ● 해설
* 4\~6: x, y, z 공간 값 설정
* 10: 1차원 배열 포인터와 3차원 배열 포인터 변수 선언
* 14: 3차원 배열 전체 공간 확보
* 17: 2차원 배열의 첫 번째 주소를 저장할 포인터 공간 할당
* 20: 다시 2차원 배열의 공간에 대한 포인터를 할당하고
* 23\~24: 그 공간에 다시 1차원 공간을 할당한다.
* 29\~38: 2차원 배열의 행의 시작 주소와 각 행간의 간격을 출력한다. 행과 행간의 간격은 16 즉, x축의 크기라고 할 수 있다. 전체 3차원 배열을 할당하고, 2차원 배열의 시작주소를 계산하고, 각 행렬이 어떻게 변하는지 확인할 수 있다.

#### ● 예제 07-04-5 : 함수의 포인터 이해하기
```C
1 #include <stdio.h>
2 #include <stdlib.h>
3 int (*in)( const char *format, ... );
4 int (*out)( const char *format, ... );
5 int main()
6 {
7 	int x, i, *array, *head;
8 	in = scanf;			// scanf를 in으로 지정 
9 	out = printf;		// printf를 out으로 지정 
10 	
11 	printf("배열의 크기를 입력하시오 : ");
12 	in("%d", &x);		// scanf 대신 in 사용 
13 	head = array = calloc(sizeof(int), x);
14 	for(i=0; i<x; i++)
15 		*array++ = i; 	//array[i] = i; 
16 	array = head;
17 	for(i=0; i<x; i++)
18 		out("%d ", *array++); // printf 대신 out 사용 
19 	printf("\n");
20 	free(head);
21 	free(array);
22 }
```

#### ● 해설
* 3,4: 함수의 포인터를 in, out으로 지정한다. in은 scanf를 out은 printf를 지정하였다. 즉, 함수의 이름을 바꾸어 사용하고 싶을 때, 또는 동일한 원형을 가지는 함수가 여럿이 있을 때, 이들을 이름을 바꾸어 사용할 때 이 방법을 쓸 수 있다. 그러나 크게 쓸 일은 없을 듯하다. 다음 장의 함수에서 이러한 예를 하나 더 살펴보도록 한다. scanf와 printf함수의 함수 원형은 stdio.h에 정의되어 있다. 함수의 원형이 다르면 에러가 발생하므로 정확하게 쓰도록 하자.
* 8,9: 함수를 함수 포인터에 할당한다. scanf는 in에, printf는 out에 할당한다.
* 13: 동적 메모리를 할당 받아 그 포인터를 head와 array에 대입한다. 여기서 malloc를 사용하지 않고 calloc를 사용하였다. 결과는 같지만 사용하는 방법이 좀 다르다. calloc는 매개변수가 2개인데, 첫 번째 매개변수는 기본적으로 할당할 메모리의 크기이며, 두 번째 매개변수는 첫 번째 매개변수와 같은 크기를 몇 개 할당할 것인지를 나타낸다. 따라서 여기서는 정수를 저장할 메모리를 x개만큼 할당한다는 의미이다.
* 14,15: array의 포인터 위치를 변경하면서 할당된 메모리에 데이터 저장
* 16: 다시 array의 시작 주소를 head로 설정
* 17,18: 할당된 메모리에 저장된 값들을 출력한다.

#### ● 예제 07-04-7 : 포인터 위치 정보를 이용하여 소수(prime number) 찾기
```C
1 #include <stdio.h>
2 #include <stdlib.h>
3 #define SIZE 1000
4 #define TRUE 1
5 #define FALSE 0
6 int main()
7 {
8 	char *sieve; /* the sieve:체(거름망) */
9 	char *sp, *tp;
10 	int number; // 위치를 나타내는 정수값 
11 
12 	sieve = (char *) malloc(SIZE);
13 	// Set the entire sieve to TRUE.
14 	for( sp = sieve; sp < sieve + SIZE; )
15 		*sp++ = TRUE;
16 
17 	*sieve = *(sieve+1) = FALSE;	// 0과 1은 소수가 아님
18 
19 	for( number = 0, tp = sieve; tp < sieve+SIZE; tp++, number++){
20 		while(*tp == FALSE) { // FALSE가 저장된 곳은 그냥 지나간다. 
21 			tp++;
22 			number++;
23 		}		// TRUE이면 벗어나서 작업한다. 
24 
25 		sp = tp; 	// 현재의 위치를 sp에 저장하고 작업을 시작한다. 
26 		
27 		// TRUE(현재위치)의 배수가 되는 곳은 모두 FALSE로 만든다. 
28 		while( sp += number, sp < sieve + SIZE )
29 			*sp = FALSE;
30 	}
31 
32 	// TRUE가 저장된 곳의 위치를 출력한다. 
33 	for( number = 0, sp = sieve; sp < sieve + SIZE; number++, sp++ ){
34 		if( *sp )
35 			printf( "%4d ", number );
36 	}
37 	free(sieve);
38 }
```
