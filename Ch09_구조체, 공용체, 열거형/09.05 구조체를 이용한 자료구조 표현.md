### 09.05 구조체를 이용한 자료구조 표현

구조체를 이용한 자료 구조를 어떻게 나타내는지 예제들을 통해서 살펴보자. 

다음 그림은 링크트 리스트의 예를 나타낸 것으로, head는 리스트의 처음 위치를 포인트하고 있으며, 리스트의 끝은 널이다.
데이터를 처음이나 중간, 끝에도 삽입할 수 있도록 프로그램을 작성할 수 있다. head에서만 데이터 삽입, 삭제가 가능하도록 하면 스택(Stack)이 되며, 
head에서는 삽입만 가능하고, 끝에서 삭제만 가능하다면 큐(Queue)가 된다. 설계하는 방법에 따라 다양하게 사용할 수 있으며, 양방향 링크트 리스트를 만들 수도 있다.

---
#### ● 예제 09-05-1 : 링크트 리스트(linked list)구현
```C
1 #include <stdio.h>
2 #include <stdlib.h>
3 #include <conio.h>
4 
5 struct node {
6 	int data;
7 	struct node *link;
8 };
9 
10 struct node *head=NULL;
11 
12 void insert(struct node *);
13 void print(struct node *);
14 void release(struct node *);
15 
16 int main()
17 {
18 	int x;
19 	struct node *temp=NULL;
20 	
21 	printf("음수 값을 입력하면 프로그램을 끝냅니다.\n"); 
22 	while(1) {
23 		printf("정수 데이터 입력 : ");
24 		scanf("%d", &x);
25 		if(x < 0)
26 			break;
27 		else {
28 			temp = malloc(sizeof(struct node));	// 동적 메모리 생성
29 			temp->data = x;	// 데이터 저장
30 			temp->link = NULL;	// 다음 포인터는 널
31 		}
32 		insert(temp);
33 	}
34 	// 출력하면서 할당된 메모리 영역을 해제함. 
35 	printf("\n정렬된 데이터\n");
36 	print(head);	// 리스트 출력
37 	printf("데이터가 삭제됩니다.\n");
38 	release(head);	// 생성된 메모리 모두 해제
39 }
40 
41 void insert(struct node *t) {
42 	struct node *pos;
43 	if(head == NULL) // 데이터가 하나도 없을 때 
44 		head = t;
45 	else if(head->link == NULL){ // 데이터가 하나만 있을 때 
46 		if(head->data > t->data) // 끝에 삽입 
47 			head->link = t;
48 		else { // 앞에 삽입 
49 			t->link = head;
50 			head = t;
51 		}
52 	}
53 	else { // 데이터가 두 개 이상일 때
54 		pos = head;
55 		while(pos) {
56 			if( pos == head && pos->data <= t->data ) {
57 				t->link = head; 	// 맨 앞에 삽입
58 				head = t;
59 				break;
60 			} else if( pos->link == NULL && pos->data >= t->data) {
61 				pos->link = t;	// 맨 끝에 삽입 
62 				break;
63 			} else if( pos->data>=t->data && pos->link->data < t->data) {
64 				t->link = pos->link; // 가운데 삽입 
65 				pos->link = t;
66 				break;
67 			} else		// 삽입할 위치를 찾아 다음 포인터로 이동 
68 				pos = pos -> link;
69 		}
70 	}
71 }
72 
73 void print(struct node *h) {
74 	while(h) {
75 		printf("%d ", h -> data);
76 		h = h-> link;
77 	}
78 	printf("\n");
79 }
80 
81 void release(struct node *c) {
82 	struct node *tmp;
83  	while(c) {
84 		tmp = c;
85 		c = c-> link;
86 		free(tmp);
87 	}
88 }
```

#### ● 해설
* 5~8: 구조체 node선언, 멤버로 데이터와 node의 포인터를 가진다.
* 10: head node를 선언하고 NULL을 초깃값으로 지정
* 12\~14: 사용할 함수의 원형 선언
* 22\~33: 음수가 입력될 때까지 하나씩 입력받아 링크에 순서대로 삽입한다.
* 25,26: 음수가 입력되면 끝냄.
* 27\~31: 음수가 아닌 값이 입력되면 동적으로 메모리 공간을 할당하고 입력된 데이터는 data멤버에 저장하고, 다음 링크는 NULL로 한다.
* 32: 링크트 리스트에 삽입하는 함수를 호출한다.
* 36: 리스트를 출력하는 함수를 호출한다.
* 38: 사용이 끝난 뒤 메모리를 해제하는 함수를 호출한다.
* 41\~71: 리스트에 노드를 삽입하는 함수
* 43: head가 NULL이면 리스트가 비어있으므로 현재 노드는 head가 된다.
* 45\~52: node가 하나만 리스트에 있을 때 앞에 넣을 것인지 뒤에 넣을 것인지 판단하여 해당 위치에 끼워 넣는다.
* 53\~70: node가 여러 개 존재할 때 해당위치를 찾아 가는 루프이다.
* 73\~79: 현재 노드를 head부터 끝까지 출력한다.
* 81\~88: 할당한 동적 공간을 해제한다.

#### ● 예제 09-05-2 : 링크트 리스트를 이용한 큐(queue) 구현
```C
1 #include <stdio.h>
2 #include <stdlib.h>
3 
4 struct node {
5 	int data;
6 	struct node *next;
7 };
8 
9 typedef struct node node;
10 node *head=NULL, *tail= NULL, *temp=NULL;
11 void insert(int);
12 void delete();
13 void print_queue();
14 void release();
15 
16 int main()
17 {
18 	int x, d;
19 	do {
20 		printf("Queue --- 1. 삽입, 2. 삭제, 3. 출력, 0. 끝\n Input : ");
21 		scanf("%d", &x);
22 		switch(x) {
23 			case 1:
24 				printf("\n데이터 입력 : ");
25 				scanf("%d", &d);
26 				insert(d);
27 				break;
28 			case 2:
29 				delete();
30 				break;
31 			case 3:
32 				print_queue();
33 				break;
34 			case 0:
35 				release();
36 				return 0;
37 		}
38 		printf("\n\n");
39 	} while(1);
40 }
41 
42 void insert(int x) {
43 	temp = malloc(sizeof(struct node));
44 	temp -> data = x;
45 	temp -> next = NULL;
46 	if(tail != NULL)
47 		tail -> next = temp;
48 	tail = temp;
49 	if(head == NULL)
50 		head = temp;
51 
52 	printf("\n%d가 삽입되었습니다.\n", x); 
53 	print_queue();
54 	return;
55 }
56 
57 void delete() {
58 	int x;
59 	if(head == NULL) {
60 		printf("\nQueue is empty!!!\n");
61 		return;
62 	}
63 	temp = head;
64 	head = head -> next;
65 	if( !head )
66 		tail = NULL;
67 	x = temp->data;
68 	free(temp);
69 	printf("%d가 삭제되었습니다.\n", x); 
70 	print_queue();
71 	return;
72 }
73 
74 void print_queue() {
75 	if(head == NULL) {
76 		printf("Queue is empty!\n");
77 		return;
78 	}
79 	temp = head;
80 	printf("현재 큐의 내용(Head...Tail) : "); 
81 	while(temp) {
82 		printf("%d ", temp->data);
83 		temp = temp->next;
84 	}
85 	return;
86 }
87 
88 void release() {
89 	while(head) {
90 		temp = head;
91 		head = head -> next;
92 		free(temp);
93 	}
94 }
```

#### ● 해설
* 4\~7: 구조체로 노드 선언, 데이터와 링크 포인터를 멤버로 갖는다.
* 9: 데이터 형 정의
* 10: head, tail, temp 노드 선언하고 초깃값으로 NULL을 지정.
* 11\~14: 함수 원형 선언
* 19\~39: 실행할 작업을 선택하고, 입력 값에 따라 해당 작업을 실행
* 23\~27: 큐에 삽입
* 28\~30: 큐에서 삭제
* 31\~33: 큐의 내용 출력
* 33\~36: 큐의 내용 전체 삭제 및 종료
* 42\~55: 큐에 노드 추가하는 함수. 큐는 FIFO이므로 tail에 추가한다.
* 57\~72: 큐에서 노드 하나 삭제. 큐의 head에서 하나 삭제한다.
* 74\~86: 큐에 저장된 각 노드의 데이터를 출력한다.
* 88\~94: 모든 메모리 할당된 내용을 제거한다.

#### ● 예제 09-05-3 : 구조체를 이용한 2진 트리(tree)를 구현하고 트리에 데이터를 저장할 때 정렬이 되도록 한다. 루트를 기준으로 왼쪽 노드에는 작은 값이 오른쪽 노드에는 큰 값이 저장되도록 한다.
```C
1 #include <assert.h>
2 #include <stdio.h>
3 #include <stdlib.h>
4 #include <time.h>
5 
6 #define SIZE 10
7 typedef int Data;
8 
9 struct node {
10 	Data		data;
11 	struct node	*left;
12 	struct node	*right;
13 };
14 
15 typedef	struct node	Node;
16 typedef	Node	*Btree;
17 
18 void	add_node(Btree, Btree, Data);
19 Btree	new_node(Data d1);
20 void	inorder(Btree);
21 void	release(Btree);
22 Btree	Root = NULL;
23 
24 int main() {
25 	Data    a, tree_data[SIZE];
26 	int     i;
27 	srand(time(0));
28 
29 	for (i=0; i<SIZE; i++)
30 		tree_data[i] = rand();
31 
32 	for (i=0; i<SIZE; i++)
33 		add_node(NULL, Root, tree_data[i]);
34 
35 	printf("---Inorder binary tree traversal:\n");
36 	inorder(Root);
37 	release(Root);
38 }
39 
40 /* Create a binary tree */ 
41 void add_node(Btree parent, Btree c_node, Data x) {
42 	if (Root == NULL) {
43 		Root = new_node(x);
44 		return;
45 	}
46 	else {
47 		if (c_node == NULL) {
48 			if (parent -> data >= x)
49 				parent -> left = new_node(x);
50 			else
51 		 		parent -> right = new_node(x);
52 		}
53 		else if (x >= c_node -> data )
54 			add_node(c_node, c_node -> right, x);
55 		else
56 			add_node(c_node, c_node -> left, x);
57 	}
58 	return;
59 }
60 
61 Btree new_node(Data d) {	 /* Create a new node. */ 
62 	Btree t;
63 	t = (Btree) malloc(sizeof(Node));
64 	assert(t != NULL);
65 	t -> data = d;
66 	t -> left = NULL;
67 	t -> right = NULL;
68 	return t;
69 }
70 
71 void inorder(Btree tree) {	/* In-order binary tree traversal. */ 
72 	if (tree != NULL) {
73 		inorder(tree -> left);
74 		printf("%10d", tree -> data);
75 		inorder(tree -> right);
76 	}
77 }
78 
79 void release(Btree tree) {		/* free tree */ 
80 	if (tree != NULL) {
81 		release(tree -> left);
82 		release(tree -> right);
83 		free(tree);
84 	}
85 }
```

#### ● 해설
* 7: int를 Data라는 이름으로 데이터 형 재 선언
* 9\~13: 구조체 선언, 데이터와 왼쪽, 오른쪽 노드의 포인터 멤버 선언.
* 15: 구조체 node를 Node로 새로운 형 선언
* 16: 구조체 Node의 포인터 Btree 선언
* 18\~21: 함수 원형 선언
* 22: 트리의 Root 선언
* 29\~30: 난수를 이용하여 데이터를 SIZE 수만큼 초기화 한다.
* 36: tree에 저장된 데이터를 inorder로 운행하여 출력하는 함수 호출.
* 41\~59: 데이터를 이용하여 트리를 구성한다.
* 42\~45: root가 NULL이면 루트에 데이터 노드 삽입
* 48\~49: 왼쪽 노드에 삽입
* 50\~51: 오른쪽 노드에 삽입
* 53\~54: 삽입할 노드를 오른쪽 부 트리를 찾는다.
* 55\~56: 삽입할 노드를 왼쪽 부 트리를 찾는다.
* 61\~69: 새로운 노드를 동적 메모리로 할당하여 포인터를 반환한다.
* 63: 메모리를 할당하지 못하면 에러메시지를 출력하고 끝낸다.
* 71\~77: inorder(중위운행)하는 함수, left→root→right순으로 운행하는 방식
* 79~85: 트리에 할당된 메모리 전체 해제
