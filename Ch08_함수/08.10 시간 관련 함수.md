● 예제 08-10-1 : tm 구조체 이용방법 및 각종 시간 관련함수 사용법
1 // mktime example: weekday calculator 
2 #include <stdio.h>
3 #include <time.h>
4 
5 int main ()
6 {
7 	time_t rawtime, user_time;
8 	struct tm * timeinfo;
9 	int year, month, day;
10 	char * weekday[] = { "일요일", "월요일", "화요일", "수요일", "목요일", "금요일", "토요일"};
11 	time ( &rawtime );
12 	timeinfo = localtime ( &rawtime );
13 	printf("오늘은 %d년 %d월 %d일 %s입니다\n", 1900+timeinfo->tm_year, 1+timeinfo->tm_mon,
14 			timeinfo->tm_mday, weekday[timeinfo->tm_wday]); 
15 	printf("%s", asctime(timeinfo) );
16 	printf("%s", asctime(gmtime(&rawtime ) ) );
17 	printf("%s", ctime (&rawtime) );
18 
19 	/* 사용자가 지정하는 날짜를 입력받는다. */ 
20 	printf("\n년/월/일 : ");
21 	scanf("%d%*c%d%*c%d", &year, &month, &day); 
22 
23 	/* 현재 시간을 얻어 사용자가 지정한 시간으로 바꾼다. */ 
24 	time ( &rawtime );
25 	timeinfo = localtime ( &rawtime );
26 	timeinfo->tm_year = year – 1900;	// 사용자가 입력한 년도를 지정한다.
27 	timeinfo->tm_mon = month – 1;	// 사용자자 입력한 월을 지정한다.
28 	timeinfo->tm_mday = day;		// 사용자가 입력한 일을 지정한다.
29 
30 	/* mktime함수를 이용하여 시간을 새로 만든다. */ 
31 	user_time = mktime ( timeinfo );
32 	printf ("%d년 %d월 %d일은 %s입니다.\n", year, month, day, weekday[timeinfo->tm_wday]);
33 }
34 
35 /* time.h에 정의되어 있는 tm 구조체
36 struct tm 
37 { 
38 	int	tm_sec;	-- 초: 0-59 (K&R says 0-61?) 
39 	int	tm_min;	-- 분: 0-59 
40 	int	tm_hour;	-- 24시간 표시 : 0-23 
41 	int	tm_mday;	-- 일 : 1-31 
42 	int	tm_mon;	-- 월 : january: 0-11  
43 	int	tm_year;	-- 년 : Years since 1900 
44 	int	tm_wday;	-- 요일 : Days since Sunday (0-6) 
45 	int	tm_yday;	-- 그 해의 1월 1일 부터 날짜 수 : 0-365 
46 	int	tm_isdst;	-- 썸머타임 적용 : +1 Daylight Savings Time, 0 No DST, -1 don't know 
47 }; 
48 */ 
● 해설
7: 시간 변수 선언, UTC 1970년 1월 1일 0시 0분 0초를 기준으로 하는 초 단위의 시간을 저장하는 형식이다.
8: 시간 구조체 변수 선언
11: time함수는 괄호 속의 변수(rawtime)에 현재 시간을 저장한다.
12: rawtime에 저장된 시간을 지역 시간을 기준으로 구조체로 바꾸어 timeinfo에 저장한다.
13: 구조체에 저장된 값을 출력한다. 년도는 1900년을 0으로 시작하며, 월도 1월이 0부터 시작한다.
15: asctime함수는 구조체 포인터가 가리키는 시간을 문자열로 변환한다.
16: gmtime은 현재 시간을 UTC(영국 그리니치) 시간 기준으로 구조체로 변환해 그 주소를 반환한다. 
17: ctime은 시간 변수값을 문자열로 변환한다.
21: 년/월/일을 키보드에서 입력받는다.
24: 현재시간을 구한다.
25: 구조체로 변경한다.
25-28: 얻은 시간에서 년/월/일을 입력한 값으로 바꾼다.
31: mktime함수를 이용하여 변경된 값으로 시간값을 설정한다.
32: 입력된 날짜에 해당하는 요일을 찾아 출력한다.
36-47: <time.h> 헤더 파일에 정의되어 있는 시간 구조체

● 예제 08-10-2 : 카운트다운
1 /* clock example: countdown */ 
2 #include <stdio.h>
3 #include <time.h>
4 
5 void wait ( int seconds )
6 {
7 	clock_t endwait;
8 	endwait = clock () + seconds * CLOCKS_PER_SEC ;
9 	while (clock() < endwait) {}
10 }
11 
12 int main ()
13 {
14 	int n;
15 	printf ("Starting countdown...\n");
16 	for (n=10; n>0; n--)
17 	{
18 		printf ("%d\n",n);
19 		wait (1);
20 	}
21 	printf ("FIRE!!!\n");
22 	return 0;
23 }
● 해설
7: 클록 틱을 저장할 변수 선언
8: clock함수는 프로그램이 시작될 때부터 지난 틱 수를 반환한다. CLOCKS_PER_SEC매크로는 1초가 몇 개의 틱으로 지정되어 있는지를 나타낸다. 이 식은 이 시점에서의 클록 틱 수를 계산한다.
9: 다시 클록 틱 수를 받아 1초가 지날 때까지 루프를 반복한다.
16-20: 10부터 1이 될 때까지 함수 wait를 반복한다. 즉 1초간 기다렸다가 값을 출력한다. 따라서 카운트다운하게 된다.

1 /* setlocale example */ 
2 #include <stdio.h>
3 #include <time.h>

12

15 
16 	int twice=0;
17

● 예제 08-10-6 : 밀리 초 구하기
1 /* difftime example */ 
2 #include <stdio.h>
3 #include <time.h>
4 
5 int main ()
6 {
7 	time_t start, end;
8 	char name[256];
9 	double diff;
10 	clock_t t1, t2;
11 	t1 = clock();		// 처음 시간 구하기 
12 	time (&start);
13 	printf ("당신의 이름을 입력하시오: ");
14 	gets (name);
15 	time (&end);
16 	diff = difftime (end, start);
17 	printf ("안녕 %s!\n", name);

19 
20 	t2 = clock() - t1;		// 나중시간 구해서 빼기 
21 	printf ("%.2f ms\n", (double)t2 / CLOCKS_PER_SEC);
22 	printf("%ld, %ld, %ld\n", t1, t2, CLOCKS_PER_SEC);
23 }

23. 시작 전의 클록 틱 수, 계산된 클록 틱 수, 1초당 틱의 수를 출력한다.
